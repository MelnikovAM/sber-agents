```python
async def transcribe_audio(audio_file_path: str) -> str:
    """
    –¢—Ä–∞–Ω—Å–∫—Ä–∏–±–∏—Ä—É–µ—Ç –∞—É–¥–∏–æ—Ñ–∞–π–ª –≤ —Ç–µ–∫—Å—Ç.

    Returns:
        str: –¢—Ä–∞–Ω—Å–∫—Ä–∏–±–∏—Ä–æ–≤–∞–Ω–Ω—ã–π —Ç–µ–∫—Å—Ç
    """
    if config.STT_PROVIDER == "openai":
        # –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ OpenAI Whisper API
        from openai import AsyncOpenAI
        
        client = AsyncOpenAI(api_key=config.OPENAI_API_KEY)
        
        with open(audio_file_path, "rb") as audio_file:
            transcript = await client.audio.transcriptions.create(
                model="whisper-1",
                file=audio_file,
                language="ru"  # –£–∫–∞–∑—ã–≤–∞–µ–º —Ä—É—Å—Å–∫–∏–π –¥–ª—è –ª—É—á—à–µ–≥–æ –∫–∞—á–µ—Å—Ç–≤–∞
            )
        
        return transcript.text
    
    elif config.STT_PROVIDER == "local":
        # –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ faster-whisper (–ª–æ–∫–∞–ª—å–Ω–æ)
        from faster_whisper import WhisperModel
        
        # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è –º–æ–¥–µ–ª–∏ (–º–æ–∂–µ—Ç –±—ã—Ç—å –≤—ã–Ω–µ—Å–µ–Ω–∞ –≤ –≥–ª–æ–±–∞–ª—å–Ω—É—é –ø–µ—Ä–µ–º–µ–Ω–Ω—É—é)
        model = WhisperModel(
            config.STT_MODEL,  # "small", "medium", "large-v2"
            device="cuda" if torch.cuda.is_available() else "cpu",
            compute_type="float16" if torch.cuda.is_available() else "int8"
        )
        
        segments, info = model.transcribe(audio_file_path, language="ru")
        
        # –°–æ–±–∏—Ä–∞–µ–º –≤–µ—Å—å —Ç–µ–∫—Å—Ç –∏–∑ —Å–µ–≥–º–µ–Ω—Ç–æ–≤

        return text.strip()
    
    else:
        raise ValueError(f"Unknown STT provider: {config.STT_PROVIDER}")
```
